Reactзҡ„context

Context ж„ҸдёәдёҠдёӢж–ҮпјҢжҸҗдҫӣдәҶдёҖдёӘж— йңҖдёәжҜҸеұӮз»„д»¶жүӢеҠЁж·»еҠ  propsпјҢе°ұиғҪеңЁз»„д»¶ж ‘й—ҙиҝӣиЎҢж•°жҚ®дј йҖ’зҡ„ж–№жі•гҖӮ

еҲӣе»әContext React.createContext(defaultValue)

import React from "react";
export const AppContext = React.createContext({
  age: 18
});

жіЁж„ҸпјҡReact.createContext(defaultValue) еҸӘжңүеҪ“ProviderдёҚеӯҳеңЁж—¶пјҢdefaultValueжүҚдјҡз”ҹж•Ҳ

import "./styles.css";
import React, { useState } from "react";
import Header from "./components/header";
import Content from "./components/content";
import TextContext from "./components/testContext";
import { AppContext } from "./common/context";

export default function App() {
  return (
    <>
      <AppContext.Provider
        value={{
          username: "е°ҸзҺӢ",
          age: 1
        }}
      >
        <div>
          <Header></Header>
          <Content></Content>
        </div>
      </AppContext.Provider>
      <TextContext></TextContext> <!-- иҝҷдёӘз»„д»¶зҡ„defaultValueжүҚдјҡз”ҹж•Ҳ -->
    </>
  );
}

Context.Provider

<AppContext.Provider value={/* жҹҗдёӘеҖј */}>

жҜҸдёӘ Context еҜ№иұЎйғҪдјҡиҝ”еӣһдёҖдёӘ Provider React з»„д»¶пјҢе®ғе…Ғи®ёж¶Ҳиҙ№з»„д»¶и®ўйҳ… context зҡ„еҸҳеҢ–гҖӮ

Provider жҺҘж”¶дёҖдёӘ value еұһжҖ§пјҢдј йҖ’з»ҷж¶Ҳиҙ№з»„д»¶гҖӮдёҖдёӘ Provider еҸҜд»Ҙе’ҢеӨҡдёӘж¶Ҳиҙ№з»„д»¶жңүеҜ№еә”е…ізі»гҖӮеӨҡдёӘ Provider д№ҹеҸҜд»ҘеөҢеҘ—дҪҝз”ЁпјҢйҮҢеұӮзҡ„дјҡиҰҶзӣ–еӨ–еұӮзҡ„ж•°жҚ®гҖӮ

еҪ“ Provider зҡ„ value еҖјеҸ‘з”ҹеҸҳеҢ–ж—¶пјҢе®ғеҶ…йғЁзҡ„жүҖжңүж¶Ҳиҙ№з»„д»¶йғҪдјҡйҮҚж–°жёІжҹ“гҖӮProvider еҸҠе…¶еҶ…йғЁ consumer з»„д»¶йғҪдёҚеҸ—еҲ¶дәҺ shouldComponentUpdate еҮҪж•°пјҢеӣ жӯӨеҪ“ consumer з»„д»¶еңЁе…¶зҘ–е…Ҳз»„д»¶йҖҖеҮәжӣҙж–°зҡ„жғ…еҶөдёӢд№ҹиғҪжӣҙж–°гҖӮ

йҖҡиҝҮж–°ж—§еҖјжЈҖжөӢжқҘзЎ®е®ҡеҸҳеҢ–пјҢдҪҝз”ЁдәҶдёҺ Object.is зӣёеҗҢзҡ„з®—жі•гҖӮ

Context.displayName

context еҜ№иұЎжҺҘеҸ—дёҖдёӘеҗҚдёә displayName зҡ„ propertyпјҢзұ»еһӢдёәеӯ—з¬ҰдёІгҖӮReact DevTools дҪҝз”ЁиҜҘеӯ—з¬ҰдёІжқҘзЎ®е®ҡ context иҰҒжҳҫзӨәзҡ„еҶ…е®№гҖӮ

зӨәдҫӢпјҢдёӢиҝ°з»„д»¶еңЁ DevTools дёӯе°ҶжҳҫзӨәдёә MyDisplayNameпјҡ

const MyContext = React.createContext(/* some value */);
MyContext.displayName = 'MyDisplayName';
<MyContext.Provider> // "MyDisplayName.Provider" еңЁ DevTools дёӯ
<MyContext.Consumer> // "MyDisplayName.Consumer" еңЁ DevTools дёӯ
